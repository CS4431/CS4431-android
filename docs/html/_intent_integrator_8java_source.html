180  public final AlertDialog initiateScan(Collection<String> desiredBarcodeFormats) {
181  Intent intentScan = new Intent(BS_PACKAGE + ".SCAN");
182  intentScan.addCategory(Intent.CATEGORY_DEFAULT);
183 
184  // check which types of codes to scan for
185  if (desiredBarcodeFormats != null) {
186  // set the desired barcode types
187  StringBuilder joinedByComma = new StringBuilder();
188  for (String format : desiredBarcodeFormats) {
189  if (joinedByComma.length() > 0) {
190  joinedByComma.append(',');
191  }
192  joinedByComma.append(format);
193  }
194  intentScan.putExtra("SCAN_FORMATS", joinedByComma.toString());
195  }
196 
197  String targetAppPackage = findTargetAppPackage(intentScan);
198  if (targetAppPackage == null) {
199  return showDownloadDialog();
200  }
201  intentScan.setPackage(targetAppPackage);
202  intentScan.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
203  intentScan.addFlags(Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET);
204  attachMoreExtras(intentScan);
205  startActivityForResult(intentScan, REQUEST_CODE);
206  return null;
207  }

222  private String findTargetAppPackage(Intent intent) {
223  PackageManager pm = activity.getPackageManager();
224  List<ResolveInfo> availableApps = pm.queryIntentActivities(intent, PackageManager.MATCH_DEFAULT_ONLY);
225  if (availableApps != null) {
226  for (String targetApp : targetApplications) {
227  if (contains(availableApps, targetApp)) {
228  return targetApp;
229  }
230  }
231  }
232  return null;

286  public static IntentResult parseActivityResult(int requestCode, int resultCode, Intent intent) {
287  if (requestCode == REQUEST_CODE) {
288  if (resultCode == Activity.RESULT_OK) {
289  String contents = intent.getStringExtra("SCAN_RESULT");
290  String formatName = intent.getStringExtra("SCAN_RESULT_FORMAT");
291  byte[] rawBytes = intent.getByteArrayExtra("SCAN_RESULT_BYTES");
292  int intentOrientation = intent.getIntExtra("SCAN_RESULT_ORIENTATION", Integer.MIN_VALUE);
293  Integer orientation = intentOrientation == Integer.MIN_VALUE ? null : intentOrientation;
294  String errorCorrectionLevel = intent.getStringExtra("SCAN_RESULT_ERROR_CORRECTION_LEVEL");
295  return new IntentResult(contents,
296  formatName,
297  rawBytes,
298  orientation,
299  errorCorrectionLevel);

346  private void attachMoreExtras(Intent intent) {
347  for (Map.Entry<String,Object> entry : moreExtras.entrySet()) {
348  String key = entry.getKey();
349  Object value = entry.getValue();
350  // Kind of hacky
351  if (value instanceof Integer) {
352  intent.putExtra(key, (Integer) value);
353  } else if (value instanceof Long) {
354  intent.putExtra(key, (Long) value);
355  } else if (value instanceof Boolean) {
356  intent.putExtra(key, (Boolean) value);
357  } else if (value instanceof Double) {
358  intent.putExtra(key, (Double) value);
359  } else if (value instanceof Float) {
360  intent.putExtra(key, (Float) value);
361  } else if (value instanceof Bundle) {
362  intent.putExtra(key, (Bundle) value);
363  } else {
364  intent.putExtra(key, value.toString());
365  }
366  }
